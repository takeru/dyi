def column_widths=(widths)
  if widths.kind_of?(Array)
    @options[:column_widths] = widths.map {|width| Length.new_or_nil(width)}
  elsif widths
    @options[:column_widths] = Length.new(widths)
  else
    @options.delete(:column_widths)
  end
  point
end
