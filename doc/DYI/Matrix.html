def initialize(*args)
  case args.first
  when :translate
    @xx = @yy = 1
    @xy = @yx = 0
    @x0 = args[1]
    @y0 = args[2] || 0
  when :scale
    @xx = args[1]
    @yy = args[2] || args[1]
    @xy = @yx = @x0 = @y0 = 0
  when :rotate
    radian = args[1] * 2 * Math::PI / 360
    @xx = @yy = Math.cos(radian)
    @xy = -(@yx = Math.sin(radian))
    @x0 = @y0 = 0
  when :skewX
    @xx = @yy = 1
    @xy = Math.tan(args[1] * 2 * Math::PI / 360)
    @yx = @x0 = @y0 = 0
  when :skewY
    @xx = @yy = 1
    @yx = Math.tan(args[1] * 2 * Math::PI / 360)
    @xy = @x0 = @y0 = 0
  else
    raise ArgumentError unless args.size == 6
    @xx, @yx, @xy, @yy, @x0, @y0 = args
  end
end
